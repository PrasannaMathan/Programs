 Question:1.
Write a C program to input an array with n elements  and perform merging operation.
 1ST WAY 
(INPUT)
#include <stdio.h>

void merging(int A[], int N1, int B[], int N2)
{
    int C[20], i, j;

    for(i = 0; i <= N1 - 1; i++)
    {
        C[i] = A[i];
    }

    for(i = N1, j = 0; j <= N2 - 1; i++, j++)
    {
        C[i] = B[j];
    }

    printf("Printing resultant array:\n");
    for(i = 0; i < (N1 + N2); i++)
    {
        printf("%d ", C[i]);
    }
}

int main()
{
    int N1, N2, A[10], B[10], i;

    printf("Input N1 & N2:\n");
    scanf("%d %d", &N1, &N2);

    printf("Input Array A:\n");
    for(i = 0; i < N1; i++)
    {
        scanf("%d", &A[i]);
    }

    printf("Input Array B:\n");
    for(i = 0; i < N2; i++)
    {
        scanf("%d", &B[i]);
    }

    merging(A, N1, B, N2);

    return 0;
}

Output:
Input N1 & N2: 5 3
Input Array A: 9 8 7 6 5
Input Array B: 4 3 2 
Printing resultant array:
9 8 7 6 5 4 3 2

 3RD WAY (SORTED MERGING)

INPUT
#include <stdio.h>

void merging(int a[], int n1, int b[], int n2)
{
    int c[30], i = 0, j = 0, k = 0;

    while(i <= n1 - 1 && j <= n2 - 1)
    {
        if(a[i] > b[j])
        {
            c[k] = b[j];
            k++;
            j++;
        }
        else
        {
            c[k] = a[i];
            k++;
            i++;
        }
    }

    if(j == n2)
    {
        for(; i <= n1 - 1; i++, k++)
        {
            c[k] = a[i];
        }
    }
    else
    {
        for(; j <= n2 - 1; j++, k++)
        {
            c[k] = b[j];
        }
    }

    for(k = 0; k <= (n1 + n2) - 1; k++)
    {
        printf("%d ", c[k]);
    }
}

int main()
{
    int n1, n2, i, a[10], b[10];

    printf("Input n1 & n2:\n");
    scanf("%d %d", &n1, &n2);

    printf("Input array a:\n");
    for(i = 0; i < n1; i++)
    {
        scanf("%d", &a[i]);
    }

    printf("Input array b:\n");
    for(i = 0; i < n2; i++)
    {
        scanf("%d", &b[i]);
    }

    merging(a, n1, b, n2);

    return 0;
}
