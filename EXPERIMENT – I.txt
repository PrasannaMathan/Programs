Question 1:
Write a C program to define four different functions (add, sub, mul, div) to perform arithmetic operations on two integers.
Input:-
#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int sub(int a, int b) {
    return a - b;
}

int mul(int a, int b) {
    return a * b;
}

int divi(int a, int b) {
    return a / b;
}

int main() {
    int num1, num2;

    printf("Enter two integers: ");
    scanf("%d %d", &num1, &num2);

    printf("Addition = %d\n", add(num1, num2));
    printf("Subtraction = %d\n", sub(num1, num2));
    printf("Multiplication = %d\n", mul(num1, num2));
    printf("Division = %d\n", divi(num1, num2));

    return 0;
}
Output:
Enter two integers: 30 15
Addition = 45
Subtraction = 15
Multiplication = 450
Division = 2


Question 2:
Write a C program to find the factorial of a number using function.
INPUT
#include <stdio.h>

int fact(int n) {
    if (n == 0 || n == 1)
        return 1;
    else
        return n * fact(n - 1);
}

int main() {
    int n;

    printf("Enter an integer: ");
    scanf("%d", &n);

    printf("Factorial = %d", fact(n));

    return 0;
}

Output:
Enter an integer: 7
Factorial = 5040

Question 3:
Write a C program to check whether a number is prime or not.
Input:
#include <stdio.h>

void prime(int num) {
    int i, flag = 0;

    for (i = 2; i <= num / 2; i++) {
        if (num % i == 0) {
            flag = 1;
            break;
        }
    }

    if (flag == 0)
        printf("%d is a Prime number", num);
    else
        printf("%d is not a Prime number", num);
}

int main() {
    int n;

    printf("Enter an integer: ");
    scanf("%d", &n);

    prime(n);

    return 0;
}

Output:
Enter an integer: 7
7 is a Prime number

Question 4:
Write a C program to generate Fibonacci series up to n terms using function.
Input:
#include <stdio.h>

void fibonacci(int n) {
    int a = 0, b = 1, c, i;

    printf("%d %d ", a, b);

    for (i = 2; i < n; i++) {
        c = a + b;
        printf("%d ", c);
        a = b;
        b = c;
    }
}

int main() {
    int n;

    printf("Enter number of terms: ");
    scanf("%d", &n);

    fibonacci(n);

    return 0;
}

Output:
Enter number of terms: 5
0 1 1 2 3

Question 5:
Write a C program to find the GCD of two numbers.
Input:

#include <stdio.h>

int gcd(int a, int b) {
    while (a != b) {
        if (a > b)
            a = a - b;
        else
            b = b - a;
    }
    return a;
}

int main() {
    int a, b;

    printf("Enter two integers: ");
    scanf("%d %d", &a, &b);

    printf("GCD = %d", gcd(a, b));

    return 0;
}
Output:
Enter two integers: 6 15
GCD = 3
