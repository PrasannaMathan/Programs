1.write a c program to create a single linked list and perform the following operation 
(1) Traversing
INPUT

#include <stdio.h>
#include <stdlib.h>

struct node
{
    int data;
    struct node *link;
};

int main()
{
    struct node *head = NULL, *temp, *p;
    int n, i;

    printf("Input the number of nodes: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++)
    {
        temp = (struct node*)malloc(sizeof(struct node));

        printf("Input the node data: ");
        scanf("%d", &temp->data);

        temp->link = NULL;

        if(head == NULL)
        {
            head = temp;
        }
        else
        {
            p = head;
            while(p->link != NULL)
            {
                p = p->link;
            }
            p->link = temp;
        }
    }

    printf("Linked List elements are:\n");

    p = head;
    while(p != NULL)
    {
        printf("%d ", p->data);
        p = p->link;
    }

    return 0;
}
