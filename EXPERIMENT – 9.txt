1)Write a c program to perform insertion and deletion operation in alinear queue as a menu driven program menu contains 3 programs (i)Insertion (ii)Deletion (iii)Display
Input:-
#include <stdio.h>
#include <stdlib.h>

#define SIZE 5

int front = -1, rear = -1;

void insertion(int a[], int item);
void deletion(int a[]);
void display(int a[]);

int main()
{
    int a[SIZE], option, item;

    do
    {
        printf("\nQueue Operations:");
        printf("\n1. Insertion");
        printf("\n2. Deletion");
        printf("\n3. Display");
        printf("\n4. Exit");
        printf("\nEnter your choice: ");
        scanf("%d", &option);

        switch(option)
        {
            case 1:
                printf("Enter the element: ");
                scanf("%d", &item);
                insertion(a, item);
                break;

            case 2:
                deletion(a);
                break;

            case 3:
                display(a);
                break;

            case 4:
                exit(0);

            default:
                printf("Invalid choice\n");
                break;
        }

    } while(option != 4);

    return 0;
}

void insertion(int a[], int item)
{
    if(rear == SIZE - 1)
    {
        printf("Queue Overflow\n");
        exit(0);
    }
    else
    {
        if(front == -1)
            front = 0;

        rear = rear + 1;
        a[rear] = item;
        printf("%d inserted into the queue\n", item);
    }
}

void deletion(int a[])
{
    if(front == -1 || front > rear)
    {
        printf("Queue Underflow\n");
    }
    else
    {
        printf("Deleted element: %d\n", a[front]);
        front = front + 1;
    }
}

void display(int a[])
{
    int i;

    if(front == -1 || front > rear)
    {
        printf("Queue is empty\n");
    }
    else
    {
        printf("The queue is: ");
        for(i = front; i <= rear; i++)
        {
            printf("%d ", a[i]);
        }
        printf("\n");
    }
}

2)Write a c program to perform insertion ,deletion,and display operations in a circular queue as a menu driven program.
Input:-
#include <stdio.h>
#include <stdlib.h>

#define MAX 5

int front = -1, rear = -1;

void insertion(int a[], int item);
void deletion(int a[]);
void display(int a[]);

int main()
{
    int a[MAX], choice, item;

    do
    {
        printf("\nCircular Queue Operations:");
        printf("\n1. Insertion");
        printf("\n2. Deletion");
        printf("\n3. Display");
        printf("\n4. Exit");
        printf("\nEnter your choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1:
                printf("Enter element to insert: ");
                scanf("%d", &item);
                insertion(a, item);
                break;

            case 2:
                deletion(a);
                break;

            case 3:
                display(a);
                break;

            case 4:
                exit(0);

            default:
                printf("Invalid choice\n");
                break;
        }

    } while(choice != 4);

    return 0;
}

void insertion(int a[], int item)
{
    if((front == 0 && rear == MAX - 1) || (front == rear + 1))
    {
        printf("Circular Queue Overflow\n");
    }
    else
    {
        if(front == -1)
        {
            front = rear = 0;
        }
        else if(rear == MAX - 1)
        {
            rear = 0;
        }
        else
        {
            rear = rear + 1;
        }

        a[rear] = item;
        printf("%d inserted into circular queue\n", item);
    }
}

void deletion(int a[])
{
    if(front == -1)
    {
        printf("Circular Queue Underflow\n");
    }
    else
    {
        printf("Deleted element: %d\n", a[front]);

        if(front == rear)
        {
            front = rear = -1;
        }
        else if(front == MAX - 1)
        {
            front = 0;
        }
        else
        {
            front = front + 1;
        }
    }
}

void display(int a[])
{
    int i;

    if(front == -1)
    {
        printf("Circular Queue is empty\n");
    }
    else
    {
        printf("Circular Queue elements are: ");

        if(front <= rear)
        {
            for(i = front; i <= rear; i++)
                printf("%d ", a[i]);
        }
        else
        {
            for(i = front; i < MAX; i++)
                printf("%d ", a[i]);

            for(i = 0; i <= rear; i++)
                printf("%d ", a[i]);
        }

        printf("\n");
    }
}
